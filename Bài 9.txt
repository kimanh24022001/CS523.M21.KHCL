//TH1:
#include <iostream>
using namespace std;
int a[1000],s[1000],b[1000],n,sum=0,M;
void daycon(int i){
	for (int j=1;j<=n;j++){
		if (sum+a[j]<=M && b[j]==0){
			s[i]=a[j];
			sum=sum+a[j];
			b[j]=1;
			if (sum==M){
				for (int m=1;m<=i;m++){
					cout<<s[m]<<" ";
				}
				cout<<endl;
			}
			else{
				daycon(i+1);
			}
			b[j]=0;
			sum=sum-a[j];
		}
	}
}
int main() {
	cin>>n>>M;
	for (int i=1;i<=n;i++){
		cin>>a[i];
		b[i]=0;
	}
	daycon(1);
	return 0;
}

#include <iostream>
using namespace std;
int a[1000],s[1000],n,sum=0,M;
void daycon(int i, int k){
	for (int j=k+1;j<=n;j++){
		if (sum+a[j]<=M){
			s[i]=a[j];
			sum=sum+a[j];
			if (sum==M){
				for (int m=1;m<=i;m++){
					cout<<s[m]<<" ";
				}
				cout<<endl;
			}
			else{
				daycon(i+1,j);
			}
			sum=sum-a[j];
		}
	}
}
int main() {
	cin>>n>>M;
	for (int i=1;i<=n;i++){
		cin>>a[i];
	}
	daycon(1,0);
	return 0;
}
