int min(float a,float b){
	if (a<b)  return a;
	else   return b;
}
float timCapDiem(float a[1000],int l,int r){
	float Min=10e9;
	if((r-l)<=3){
		if (abs(a[l+1]-a[l])-abs(a[r-1]-a[r-2])<=0)
	    { return abs(a[l+1]-a[l]);}
	    else return abs(a[r-1]-a[r-2]);
	    
	}
	int m=(l+r+1)/2;
	float dist_left=timCapDiem(a,l,m);  
	float dist_right=timCapDiem(a,m+1,r);
	float dist_min=min(dist_left,dist_right);
//	cout<<dist_left<<dist_right;
	float temp[1000];
	int index=0;
	for(int i=l;i<r;i++){
		if (abs(a[m]-a[i])<=dist_min){
			temp[index]=a[i];
			index++;
		}
	}
	for (int i=0;i<index-1;i++){
		if(Min>=(temp[i+1]-temp[i])){
		    Min=temp[i+1]-temp[i];
		}
	}
	return min(dist_min,Min);
	
}
